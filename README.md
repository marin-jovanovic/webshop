# webshop

reccomendation app based on purchase history



install

	run "npm i" in /db and /react


/db

	db api
	
/db_as_json	

	exported database in json format
	
/py 	

	util scripts for creating db
	
/react

	app for showing simple use case


run

	run db react part
		npm start
	run json db in react
		npm run server

	run db api
		npm run start
